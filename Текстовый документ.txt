        const canvas = document.getElementById('doc-system-bg');
        const ctx = canvas.getContext('2d');
        let w, h;
        function resize() { w = canvas.width = window.innerWidth; h = canvas.height = window.innerHeight; }

        class DocBlock {
            constructor() { this.init(); }
            init() {
                this.w = 150 + Math.random()*100; this.h = 40 + Math.random()*40;
                this.x = Math.random()*(w-this.w); this.y = Math.random()*(h-this.h);
                this.life=0; this.maxLife=200+Math.random()*200; this.opacity=0; this.state='in';
            }
            update() {
                if(this.state==='in') { this.opacity+=0.01; if(this.opacity>=0.5) this.state='hold'; }
                else if(this.state==='hold') { this.life++; if(this.life>this.maxLife) this.state='out'; }
                else { this.opacity-=0.01; if(this.opacity<=0) this.init(); }
            }
            draw() {
                ctx.fillStyle=`rgba(0,33,71,${0.05*this.opacity})`; ctx.strokeStyle=`rgba(0,33,71,${0.1*this.opacity})`;
                ctx.beginPath(); ctx.roundRect(this.x,this.y,this.w,this.h,4); ctx.fill(); ctx.stroke();
                ctx.fillStyle=`rgba(0,33,71,${0.1*this.opacity})`;
                ctx.fillRect(this.x+10,this.y+10,this.w-20,4); ctx.fillRect(this.x+10,this.y+20,this.w*0.6,4);
            }
        }
        const blocks = Array.from({length:5}, ()=>new DocBlock());
        function animate() {
            ctx.clearRect(0,0,w,h);
            ctx.strokeStyle='rgba(0,33,71,0.03)'; ctx.lineWidth=1; ctx.beginPath();
            for(let x=0;x<w;x+=60){ctx.moveTo(x,0);ctx.lineTo(x,h);}
            for(let y=0;y<h;y+=60){ctx.moveTo(0,y);ctx.lineTo(w,y);}
            ctx.stroke();
            blocks.forEach(b=>{b.update();b.draw();});
            requestAnimationFrame(animate);
        }
        animate();
